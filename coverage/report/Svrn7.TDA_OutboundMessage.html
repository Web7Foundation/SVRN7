 327    /// <summary>
 328    /// Creates a dedicated isolated <see cref="Runspace"/>, optionally imports a
 329    /// JIT LOBE module into it, then invokes the cmdlet or agent script pipeline.
 330    /// The runspace is disposed on return — a fault here cannot affect any other
 331    /// concurrent dispatch.
 332    ///
 333    /// Pipeline pattern (PowerShell, pass-by-reference):
 334    ///   Dequeue-Svrn7Message -Did $did | Invoke-{Lobe}Cmdlet -MessageDid $did
 335    /// </summary>
 336    private async Task InvokeCmdletPipelineAsync(
 337        string cmdletOrScript, string modulePath, string didUrl, CancellationToken ct)
 338    {
 339        using var isolated = _pool.CreateIsolatedPipeline();
 340        var ps = isolated.Ps;
 341
 342        // For LOBE cmdlets (not agent .ps1 scripts), ensure the module is present
 343        // in this runspace. Eager LOBEs are skipped (already in the ISS).
 344        // JIT LOBEs are imported now into this dedicated runspace.
 345        if (!cmdletOrScript.EndsWith(".ps1", StringComparison.OrdinalIgnoreCase))
 346            await _lobes.EnsureLoadedAsync(ps, modulePath, ct);
 347
 348        ps.Commands.Clear();
 349        if (cmdletOrScript.EndsWith(".ps1", StringComparison.OrdinalIgnoreCase))
 350        {
 351            // Agent script: executed with MessageDid parameter.
 352            ps.AddCommand(cmdletOrScript)
 353              .AddParameter("MessageDid", didUrl);
 354        }
 355        else
 356        {
 357            // LOBE cmdlet pipeline: Dequeue-Svrn7Message | cmdlet (pass-by-reference).
 358            ps.AddCommand("Dequeue-Svrn7Message")
 359              .AddParameter("Did", didUrl)
 360              .AddStatement()
 361              .AddCommand(cmdletOrScript)
 362              .AddParameter("MessageDid", didUrl);
 363        }
 364
 365        using var invokeActivity = Svrn7Telemetry.Source.StartActivity(
 366            Svrn7Telemetry.ActivityInvoke,
 367            ActivityKind.Internal);
 368        invokeActivity?.SetTag(Svrn7Telemetry.TagMessageId,      didUrl)
 369                       .SetTag(Svrn7Telemetry.TagLobeEntrypoint, cmdletOrScript);
 370
 371        _log.LogTrace("PS invoke: {Cmdlet} -MessageDid {Did}", cmdletOrScript, didUrl);
 372
 373        // ps.Invoke() is synchronous. Wrap in Task.Run so it doesn't block the thread pool.
 374        // PowerShell does not honour CancellationToken internally — interruption is via
 375        // ps.Stop(). Apply an external timeout using WaitAsync + a linked CTS.
 376        var invokeTask = Task.Run(() => ps.Invoke());
 377
 378        if (_opts.LobeInvocationTimeoutSeconds > 0)
 379        {
 380            using var timeoutCts = CancellationTokenSource.CreateLinkedTokenSource(ct);
 381            timeoutCts.CancelAfter(TimeSpan.FromSeconds(_opts.LobeInvocationTimeoutSeconds));
 382            try
 383            {
 384                await invokeTask.WaitAsync(timeoutCts.Token);
 385            }
 386            catch (OperationCanceledException) when (!ct.IsCancellationRequested)
 387            {
 388                // Timeout fired (not shutdown) — stop the runspace and fail the message.
 389                ps.Stop();
 390                try { await invokeTask.WaitAsync(TimeSpan.FromSeconds(5), CancellationToken.None); }
 391                catch { /* best-effort wind-down; runspace disposed by IsolatedPipeline */ }
 392                throw new TimeoutException(
 393                    $"LOBE cmdlet '{cmdletOrScript}' timed out after {_opts.LobeInvocationTimeoutSeconds}s for message {
 394            }
 395        }

 504        {
 505            try
 506            {
 507                using var content = new System.Net.Http.StringContent(
 508                    msg.PackedMessage,
 509                    System.Text.Encoding.UTF8,
 510                    "application/didcomm-encrypted+json");

 513
 514                if (response.IsSuccessStatusCode)
 515                {

 517                        "Switchboard: outbound delivered to {Endpoint} ({Status}).",
 518                        endpoint, (int)response.StatusCode);
 519                    deliverActivity?.SetTag(Svrn7Telemetry.TagOutcome, "delivered")
 520                                    .SetStatus(ActivityStatusCode.Ok);
 521                    return; // success
 522                }

 524                // Non-success HTTP status — treat as a retryable failure.
 525                lastException = new HttpRequestException(

 529                    (int)response.StatusCode, attempt, OutboundMaxAttempts, endpoint);
